Developer Quick Start

Integrate Ocrypt into your application in minutes. Add distributed tryst cryptography to protect your users' data with just a few lines of code.

What is Ocrypt?

Ocrypt is OpenADP's high-level API that makes distributed cryptography as simple as traditional password-based encryption. Instead of storing encryption keys on a single server (vulnerable to government pressure), Ocrypt distributes key shares across multiple independent servers in different countries.

Integration Guide

1

Setup OpenADP JavaScript SDK

Clone the OpenADP repository and navigate to the JavaScript SDK:

git clone https://github.com/waywardgeek/openadp.git cd openadp/sdk/javascript npm install
2

Import Ocrypt

import { register, recover } from './path/to/openadp/sdk/javascript/src/ocrypt.js';
3

Register User Data

When a user creates an account or wants to back up sensitive data:

async function backupUserData(userID, sensitiveData, userPIN) { try { // Convert sensitive data to bytes const dataBytes = new TextEncoder().encode(JSON.stringify(sensitiveData)); // Register with OpenADP - this distributes the encryption key const metadata = await register( userID, // Unique user identifier 'my-app', // Your application name dataBytes, // The sensitive data to protect userPIN, // User's PIN/password 'current', // Backup version 3 // Maximum wrong PIN attempts ); // Store metadata locally - this is safe to store anywhere localStorage.setItem(`backup_${userID}`, JSON.stringify(metadata)); console.log('‚úÖ User data backed up successfully!'); return metadata; } catch (error) { console.error('‚ùå Backup failed:', error.message); throw error; } }
4

Recover User Data

When a user needs to restore their data:

async function restoreUserData(userID, userPIN) { try { // Get metadata from local storage const metadataStr = localStorage.getItem(`backup_${userID}`); if (!metadataStr) { throw new Error('No backup found for this user'); } const metadata = JSON.parse(metadataStr); // Recover data from OpenADP network const recoveredBytes = await recover(metadata, userPIN); // Convert back to original data const recoveredData = JSON.parse(new TextDecoder().decode(recoveredBytes)); console.log('‚úÖ User data recovered successfully!'); return recoveredData; } catch (error) { console.error('‚ùå Recovery failed:', error.message); throw error; } }

Complete Example

Here's a complete working example:

// Example: Secure Notes App class SecureNotesApp { constructor() { this.userID = 'user123'; } async saveNote(noteText, userPIN) { const noteData = { content: noteText, timestamp: new Date().toISOString(), version: '1.0' }; try { const metadata = await register( this.userID, 'secure-notes', new TextEncoder().encode(JSON.stringify(noteData)), userPIN, 'current', 3 ); // Store metadata - safe to store anywhere localStorage.setItem('note_metadata', JSON.stringify(metadata)); alert('‚úÖ Note saved securely!'); } catch (error) { alert('‚ùå Failed to save note: ' + error.message); } } async loadNote(userPIN) { try { const metadata = JSON.parse(localStorage.getItem('note_metadata')); const recoveredBytes = await recover(metadata, userPIN); const noteData = JSON.parse(new TextDecoder().decode(recoveredBytes)); return noteData.content; } catch (error) { alert('‚ùå Failed to load note: ' + error.message); return null; } } } // Usage const app = new SecureNotesApp(); // Save a note await app.saveNote('My secret note content', '1234'); // Load the note const noteContent = await app.loadNote('1234'); console.log('Recovered note:', noteContent);

Register User Data

func backupUserData(userID string, sensitiveData interface{}, userPIN string) ([]byte, error) { // Convert data to bytes dataBytes, err := json.Marshal(sensitiveData) if err != nil { return nil, fmt.Errorf("failed to marshal data: %v", err) } // Register with OpenADP metadata, err := ocrypt.Register( userID, // User ID "my-app", // Application name dataBytes, // Sensitive data userPIN, // User PIN "current", // Version 3, // Max wrong attempts ) if err != nil { return nil, fmt.Errorf("registration failed: %v", err) } // Store metadata (safe to store anywhere) metadataBytes, err := json.Marshal(metadata) if err != nil { return nil, fmt.Errorf("failed to marshal metadata: %v", err) } err = ioutil.WriteFile(fmt.Sprintf("backup_%s.json", userID), metadataBytes, 0644) if err != nil { return nil, fmt.Errorf("failed to save metadata: %v", err) } fmt.Println("‚úÖ User data backed up successfully!") return metadataBytes, nil }
4

Recover User Data

func restoreUserData(userID string, userPIN string) (interface{}, error) { // Load metadata metadataBytes, err := ioutil.ReadFile(fmt.Sprintf("backup_%s.json", userID)) if err != nil { return nil, fmt.Errorf("failed to load metadata: %v", err) } var metadata map[string]interface{} err = json.Unmarshal(metadataBytes, &metadata) if err != nil { return nil, fmt.Errorf("failed to unmarshal metadata: %v", err) } // Recover data from OpenADP network recoveredBytes, err := ocrypt.Recover(metadata, userPIN) if err != nil { return nil, fmt.Errorf("recovery failed: %v", err) } // Convert back to original data var recoveredData interface{} err = json.Unmarshal(recoveredBytes, &recoveredData) if err != nil { return nil, fmt.Errorf("failed to unmarshal recovered data: %v", err) } fmt.Println("‚úÖ User data recovered successfully!") return recoveredData, nil }
